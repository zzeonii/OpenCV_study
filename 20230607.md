# 파이썬 가상환경 만들기
- $ sudo apt-get update && sudo apt-get upgrade
- 파이썬 버전확인
  - $ python3 -V
- pip 버전확인
  - $ pip -V
- pip3 설치
  - $ sudo apt install python-pip
- venv 설치확인
  - $ python3 -m venv -h
- 가상환경 설치 (파이썬 버전에 맞춰서 ex 3.8)
  - $ sudo apt install python3.8-venv
- 가상환경 생성 python3 -m venv [가상환경이름]
  - $ python3 -m venv venv
- 가상환경 활성
- linux or macOS
- . = 현재위치 | .. = 상위폴더
  - $ source ./venv/bin/activate
- 가상환경 확인
  - (venv)$ python -V 
  - (venv)$ pip -V
- 가상환경 비활성
  - (venv)$ deactivate
